bigsrd2<- bigsrd %>% mutate(Player=gsub("\\\\.*","",Player))
 bigsrd3 <- bigsrd2 %>% filter(G > 41) %>%

  arrange(desc(PER))
bigsrd4 <-bigsrd3 %>% filter(MP > 1380)
ggplot() + geom_point(data=bigsrd4, aes(x=PER, y=WS))

ggplot() +

    aes(x=PER, y=VORP, size=WS)

  ungroup() %>% 
  summarise(
    offense = mean(PER), 
    defense = mean(VORP)
